//XOR ZEUG

#include <stdio.h>
#include <string.h>
void encryptDecrypt(char inpString[])
{
    char xorKey = '1';
    int len = strlen(inpString);
    for (int i = 0; i < len; i++)
    {
        inpString[i] = inpString[i] ^ xorKey;
        printf("%c",inpString[i]);
    }
}

int main()
{
    char sampleString[] = "Mein Name is Leon\nHallo mein name ist auch Leon und heuzte";
    printf("Encrypted String: ");
    encryptDecrypt(sampleString);
    printf("\n");
    printf("Decrypted String: ");
    encryptDecrypt(sampleString);

    return 0;
}

//PQ 
#include <stdio.h>
#include <math.h>

int main(){
    double a, p, q, sum, sum2;
    printf("a, p, q eingeben / jeweils mit Leerzeie trennen:\n");
    scanf("%lf %lf %lf", &a, &p, &q);
    p = p / a;
    q = q / a;
    sum = -(p * 0.5) + sqrt((p * 0.5) * (p * 0.5) - q);
    sum2 = -(p * 0.5) - sqrt((p * 0.5) * (p * 0.5) - q);
    printf("Ergebnis: ");
    printf("%lf, %lf", sum, sum2);
    return 0;
}
